SELECT
    d.year_num AS claim_year,
    d.month_num AS claim_month,
    SUM(f.billed_amount) AS total_billed,
    SUM(f.paid_amount) AS total_paid,
    COUNT(*) AS total_claims
FROM fact_claims f
JOIN dim_date d
    ON f.date_key = d.date_key
GROUP BY
    d.year_num,
    d.month_num
ORDER BY
    d.year_num,
    d.month_num;
